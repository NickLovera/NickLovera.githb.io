static int jumpingOnClouds(int[] c, int k) {
    int place = 0,energy = 100;
    while(true){
        place = (place+k) % c.length;
        energy--;
        if(c[place] == 1){
            energy -= 2;
        }
                
        if(place == 0){
            break;
        }
    }
    return energy;
}
